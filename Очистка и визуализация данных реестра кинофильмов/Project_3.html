import re
def disk_func(l):
    if re.search('HDD',l):  return 'HDD'
    elif re.search('BETA[KC]AM',l):  return 'BETACAM'
    elif re.search('DVD',l):  return 'DVD'
    elif re.search('DP',l):  return 'DPX'
    elif re.search('DCP',l):  return 'DCP'
    else: return  ''
